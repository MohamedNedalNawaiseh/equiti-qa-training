Advanced Page Object Model

Base Page Class

// pages/base/BasePage.ts
import { Page, Locator, expect } from '@playwright/test';

export abstract class BasePage {
  protected readonly page: Page;
  protected readonly baseURL: string;

  constructor(page: Page, baseURL: string = '') {
    this.page = page;
    this.baseURL = baseURL;
  }

  async goto(path: string): Promise {
    await this.page.goto(`${this.baseURL}${path}`);
    await this.waitForPageLoad();
  }

  async waitForPageLoad(): Promise {
    await this.page.waitForLoadState('networkidle');
  }

  async waitForURL(urlPattern: string | RegExp): Promise {
    await this.page.waitForURL(urlPattern, { timeout: 30000 });
  }

  protected async clickAndWaitForNavigation(
    locator: Locator,
    options?: { timeout?: number }
  ): Promise {
    await Promise.all([
      this.page.waitForNavigation(options),
      locator.click()
    ]);
  }

  protected async getByRoleText(
    role: 'button' | 'link' | 'heading' | etc,
    text: string,
    options?: { exact?: boolean }
  ): Promise {
    return this.page.getByRole(role, { name: text, ...options });
  }

  protected async assertNoErrors(): Promise {
    const errors: string[] = [];
    this.page.on('pageerror', (err) => errors.push(err.message));
    // Add console error listener if needed
    expect(errors).toHaveLength(0);
  }
}

Base Component Class

// pages/base/BaseComponent.ts
import { Locator, Page } from '@playwright/test';

export abstract class BaseComponent {
  protected readonly root: Locator;

  constructor(root: Locator) {
    this.root = root;
  }

  async isVisible(): Promise {
    return this.root.isVisible();
  }

  async waitForVisible(timeout = 5000): Promise {
    await this.root.waitFor({ state: 'visible', timeout });
  }

  async waitForHidden(timeout = 5000): Promise {
    await this.root.waitFor({ state: 'hidden', timeout });
  }

  protected getChild(selector: string): Locator {
    return this.root.locator(selector);
  }
}

// Example: Toast Component
export class ToastComponent extends BaseComponent {
  constructor(page: Page) {
    super(page.locator('.toast-container'));
  }

  getMessage(): Locator {
    return this.root.locator('.toast-message');
  }

  getCloseButton(): Locator {
    return this.root.locator('.toast-close');
  }

  async getText(): Promise {
    return this.getMessage().textContent() ?? '';
  }

  async close(): Promise {
    await this.getCloseButton().click();
    await this.waitForHidden();
  }
}

Complete Login Page Example

// pages/login/LoginPage.ts
import { Page, Locator, expect } from '@playwright/test';
import { BasePage } from '../base/BasePage';

interface LoginCredentials {
  email: string;
  password: string;
}

export class LoginPage extends BasePage {
  // Selectors
  private readonly emailInput = this.page.getByLabel('Email');
  private readonly passwordInput = this.page.getByLabel('Password');
  private readonly submitButton = this.page.getByRole('button', { name: /sign in/i });
  private readonly errorAlert = this.page.locator('[role="alert"]');
  private readonly forgotPasswordLink = this.page.locator('text="Forgot Password?"');

  // Constructor
  constructor(page: Page) {
    super(page, process.env.BASE_URL);
  }

  // Navigation
  async goto(): Promise {
    await super.goto('/login');
  }

  // Actions
  async login(credentials: LoginCredentials): Promise {
    await this.emailInput.fill(credentials.email);
    await this.passwordInput.fill(credentials.password);
    await this.submitButton.click();
  }

  async loginAsAdmin(): Promise {
    await this.login({
      email: process.env.ADMIN_EMAIL!,
      password: process.env.ADMIN_PASSWORD!
    });
  }

  async clickForgotPassword(): Promise {
    await this.forgotPasswordLink.click();
  }

  // Assertions
  async expectToBeOnLoginPage(): Promise {
    await expect(this.page).toHaveURL(/\/login$/);
    await expect(this.emailInput).toBeVisible();
  }

  async expectErrorMessage(message: string): Promise {
    await expect(this.errorAlert).toHaveText(message);
  }

  async expectErrorToBeVisible(): Promise {
    await expect(this.errorAlert).toBeVisible();
  }

  // Validations
  async getEmailValidationError(): Promise {
    const errorLocator = this.emailInput.locator('..').locator('.form-error');
    if (await errorLocator.isVisible()) {
      return errorLocator.textContent();
    }
    return null;
  }
}

API Helper

// helpers/api.helper.ts
import { request, APIRequestContext } from '@playwright/test';

export class ApiHelper {
  private context: APIRequestContext;
  private baseURL: string;
  private authToken?: string;

  constructor(baseURL: string) {
    this.baseURL = baseURL;
  }

  async init(authToken?: string): Promise {
    this.context = await request.newContext({
      baseURL: this.baseURL,
      extraHTTPHeaders: authToken 
        ? { 'Authorization': `Bearer ${authToken}` }
        : {}
    });
    this.authToken = authToken;
  }

  async get(endpoint: string): Promise {
    const response = await this.context.get(endpoint);
    if (!response.ok()) {
      throw new Error(`GET ${endpoint} failed: ${response.status()} ${await response.text()}`);
    }
    return response.json();
  }

  async post(endpoint: string, data: object): Promise {
    const response = await this.context.post(endpoint, { data });
    if (!response.ok()) {
      throw new Error(`POST ${endpoint} failed: ${response.status()} ${await response.text()}`);
    }
    return response.json();
  }

  async put(endpoint: string, data: object): Promise {
    const response = await this.context.put(endpoint, { data });
    if (!response.ok()) {
      throw new Error(`PUT ${endpoint} failed: ${response.status()} ${await response.text()}`);
    }
    return response.json();
  }

  async delete(endpoint: string): Promise {
    const response = await this.context.delete(endpoint);
    if (!response.ok()) {
      throw new Error(`DELETE ${endpoint} failed: ${response.status()}`);
    }
  }

  async cleanup(): Promise {
    await this.context.dispose();
  }
}

Test Data Builder

// data/user.builder.ts
interface UserData {
  id?: string;
  name: string;
  email: string;
  role: 'admin' | 'user' | 'guest';
  createdAt?: Date;
}

class UserBuilder {
  private data: Partial = {
    role: 'user'
  };

  withName(name: string): this {
    this.data.name = name;
    return this;
  }

  withEmail(email: string): this {
    this.data.email = email;
    return this;
  }

  withRole(role: UserData['role']): this {
    this.data.role = role;
    return this;
  }

  withId(id: string): this {
    this.data.id = id;
    return this;
  }

  build(): UserData {
    if (!this.data.name || !this.data.email) {
      throw new Error('Name and email are required');
    }
    return {
      id: this.data.id ?? `user_${Date.now()}`,
      name: this.data.name,
      email: this.data.email,
      role: this.data.role!,
      createdAt: new Date()
    };
  }

  static random(): UserBuilder {
    const random = Math.random().toString(36).substr(2, 9);
    return new UserBuilder()
      .withName(`Test User ${random}`)
      .withEmail(`test_${random}@example.com`);
  }
}

// Usage
const user = UserBuilder.random()
  .withRole('admin')
  .build();

Custom Reporter

// reporters/custom.reporter.ts
import {
  Reporter,
  FullConfig,
  Suite,
  TestCase,
  TestResult,
  FullResult
} from '@playwright/test/reporter';

class CustomReporter implements Reporter {
  private results: { passed: number; failed: number; skipped: number } = {
    passed: 0,
    failed: 0,
    skipped: 0
  };

  onTestEnd(test: TestCase, result: TestResult): void {
    switch (result.status) {
      case 'passed':
        this.results.passed++;
        break;
      case 'failed':
        this.results.failed++;
        console.log(`❌ FAILED: ${test.title}`);
        console.log(`   Error: ${result.errors[0]?.message}`);
        break;
      case 'skipped':
        this.results.skipped++;
        break;
    }
  }

  onEnd(result: FullResult): void {
    console.log('\n═══════════════════════════════════════');
    console.log('         TEST EXECUTION SUMMARY        ');
    console.log('═══════════════════════════════════════');
    console.log(`   ✅ Passed: ${this.results.passed}`);
    console.log(`   ❌ Failed: ${this.results.failed}`);
    console.log(`   ⏭️  Skipped: ${this.results.skipped}`);
    console.log('═══════════════════════════════════════\n');
  }
}

export default CustomReporter;

Enterprise Best Practices

1. Test Independence

Each test should be able to run independently. Use fixtures for setup/teardown.

2. Meaningful Names

Test names should describe what they verify: "should_show_error_when_email_invalid"

3. Single Responsibility

Each test should verify one behavior. Multiple assertions are OK if related.

4. Data Cleanup

Clean up test data after tests. Use beforeEach/afterEach for isolation.

5. Environment Config

Use environment variables for all environment-specific values.

6. Error Context

Include enough context in assertions to diagnose failures quickly.

Anti-Patterns to Avoid

// ❌ BAD PRACTICES

// 1. Using sleep
await page.waitForTimeout(5000); // Never do this!

// 2. Using force: true (bypasses actionability checks)
await button.click({ force: true }); // Can cause flaky tests

// 3. Hardcoded values in tests
const userId = '123'; // Should come from test data

// 4. Not cleaning up state
// Leaving test data pollutes subsequent tests

// 5. Complex XPath expressions
await page.locator('//div[1]/div[2]/div[3]/span').click(); // Brittle!

// 6. Not handling loading states
// Tests fail intermittently without proper waits

// ✅ GOOD PRACTICES

// 1. Use built-in waits
await expect(locator).toBeVisible();

// 2. Use semantic selectors
await page.getByRole('button', { name: 'Submit' }).click();

// 3. Use test data factories
const user = UserBuilder.random().build();

// 4. Clean up in afterEach
afterEach(async ({ page }) => {
  await cleanupTestData(page);
});

// 5. Use Page Objects
await loginPage.login(credentials);

// 6. Add retry logic for known flakiness
test.setTimeout(60000); // Increase timeout for slow tests
