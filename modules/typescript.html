TypeScript for QA Engineers

Learn TypeScript from scratch - essential for Playwright automation

What is TypeScript?

TypeScript is a strongly-typed programming language that builds on JavaScript. It adds type safety, making code more reliable and easier to debug - perfect for test automation.

Why TypeScript for QA?

  • Catch Errors Early: Types are checked at compile time, not runtime
  • Better IDE Support: Autocomplete, refactoring, navigation
  • Playwright Native Support: Playwright is written in TypeScript
  • Maintainable Code: Self-documenting code with types
  • Industry Standard: Most modern test frameworks use TypeScript

TypeScript vs JavaScript

// JavaScript (no types)
function login(email, password) {
  return fetch('/api/login', {
    method: 'POST',
    body: JSON.stringify({ email, password })
  });
}

// TypeScript (with types)
function login(email: string, password: string): Promise {
  return fetch('/api/login', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ email, password })
  });
}

Variables & Basic Types

Declaring Variables

// let - mutable variable (prefer this in tests)
let count = 0;
count = 5; // OK

// const - immutable (use when value won't change)
const PI = 3.14159;
const BASE_URL = 'https://api.example.com';
// PI = 3; // Error!

// var - old style (avoid in new code)
var name = 'old'; // Don't use

Basic Types

// String
let username: string = 'testuser';
let greeting: string = `Hello, ${username}`; // Template literal

// Number
let age: number = 25;
let price: number = 19.99;
let percent: number = 100; // integers and decimals are both number

// Boolean
let isLoggedIn: boolean = true;
let hasAccess: boolean = false;

// Array
let scores: number[] = [95, 87, 92];
let names: string[] = ['Ali', 'Sara', 'Omar'];
let mixed: Array = [1, 2, 3]; // Generic syntax

// Tuple (fixed-length array with types)
let user: [string, number] = ['Ahmed', 30];
// user = [30, 'Ahmed']; // Error! Order matters

// Any (avoid when possible - defeats type checking)
let dynamic: any = 'could be anything';
dynamic = 123; // OK
dynamic = true; // OK

// Void (no return value)
function logMessage(message: string): void {
  console.log(message);
}

// Null and Undefined
let empty: null = null;
let notDefined: undefined = undefined;

Type Inference

// TypeScript infers types when not explicitly declared
let inferred = 'hello'; // TypeScript knows this is string
inferred = 123; // Error! Type is inferred as string

// But you can still declare types explicitly
let explicit: string = 'hello';
explicit = 123; // Error!

Functions

Function Declarations

// Basic function with types
function add(a: number, b: number): number {
  return a + b;
}

// Arrow function (preferred in modern code)
const multiply = (a: number, b: number): number => {
  return a * b;
};

// Short arrow function (implicit return)
const square = (x: number): number => x * x;

// Optional parameters
function greet(name: string, greeting?: string): string {
  return greeting ? `${greeting}, ${name}!` : `Hello, ${name}!`;
}
greet('Ahmed');           // "Hello, Ahmed!"
greet('Ahmed', 'Hi');     // "Hi, Ahmed!"

// Default parameters
function createUser(name: string, role: string = 'user'): object {
  return { name, role };
}

// Rest parameters
function sum(...numbers: number[]): number {
  return numbers.reduce((total, n) => total + n, 0);
}
sum(1, 2, 3, 4, 5); // 15

Function Types

// Function type annotation
let myFunc: (x: number, y: number) => number;

// Using in practice
type Operation = (a: number, b: number) => number;

const add: Operation = (a, b) => a + b;
const subtract: Operation = (a, b) => a - b;

// Callback functions
function processData(
  data: string[],
  callback: (item: string, index: number) => string
): string[] {
  return data.map(callback);
}

const upperCase = processData(
  ['apple', 'banana', 'cherry'],
  (item, index) => item.toUpperCase()
);
// Result: ['APPLE', 'BANANA', 'CHERRY']

Arrays & Objects

Array Methods

const numbers = [1, 2, 3, 4, 5];

// map - transform each item
const doubled = numbers.map(n => n * 2); // [2, 4, 6, 8, 10]

// filter - keep items that pass test
const evens = numbers.filter(n => n % 2 === 0); // [2, 4]

// find - first item that passes test
const firstEven = numbers.find(n => n % 2 === 0); // 2

// reduce - combine into single value
const sum = numbers.reduce((total, n) => total + n, 0); // 15

// some/every - boolean checks
const hasEven = numbers.some(n => n % 2 === 0); // true
const allPositive = numbers.every(n => n > 0); // true

// Chaining
const result = numbers
  .filter(n => n > 2)
  .map(n => n * 10)
  .reduce((a, b) => a + b, 0); // 120 (3+4+5 = 12 * 10 = 120

Objects

// Object type declaration
const user: {
  name: string;
  age: number;
  email?: string; // optional
  role: 'admin' | 'user' | 'guest'; // union type
} = {
  name: 'Ahmed',
  age: 30,
  role: 'user'
};

// Accessing properties
console.log(user.name); // 'Ahmed'
console.log(user['age']); // 30

// Modifying (only if not const)
user.age = 31; // OK
// user.role = 'superadmin'; // Error! Not in union type

// Object destructuring
const { name, age, role } = user;
console.log(name); // 'Ahmed'

// Spread operator
const updatedUser = { ...user, age: 32, email: 'ahmed@example.com' };
// { name: 'Ahmed', age: 32, role: 'user', email: 'ahmed@example.com' }

Useful Object Methods

const user = { name: 'Ahmed', age: 30, city: 'Amman' };

// Object.keys, values, entries
Object.keys(user);    // ['name', 'age', 'city']
Object.values(user);  // ['Ahmed', 30, 'Amman']
Object.entries(user); // [['name', 'Ahmed'], ['age', 30], ['city', 'Amman']]

// Loop through entries
Object.entries(user).forEach(([key, value]) => {
  console.log(`${key}: ${value}`);
});

// Create object from entries
const parsed = Object.fromEntries([['a', 1], ['b', 2]]);
// { a: 1, b: 2 }

Interfaces

Interfaces define the structure of objects. They're essential for creating type-safe test utilities and page objects.

Basic Interfaces

// Define interface
interface User {
  id: number;
  name: string;
  email: string;
  role: 'admin' | 'user' | 'guest';
  createdAt?: Date; // optional
}

// Use interface
const user: User = {
  id: 1,
  name: 'Ahmed Hassan',
  email: 'ahmed@example.com',
  role: 'user'
};

// Function with interface parameter
function printUser(user: User): void {
  console.log(`${user.name} (${user.email}) - ${user.role}`);
}
printUser(user); // "Ahmed Hassan (ahmed@example.com) - user"

// Interface with method
interface Logger {
  log(message: string): void;
  error(message: string, error?: Error): void;
}

class ConsoleLogger implements Logger {
  log(message: string): void {
    console.log(`[LOG] ${message}`);
  }
  error(message: string, error?: Error): void {
    console.error(`[ERROR] ${message}`, error);
  }
}

Extending Interfaces

// Base interface
interface BaseEntity {

// Extend base
interface User extends BaseEntity {

const user: User = {

Interface for Test Data

// Define test data structure
interface LoginTestData {
  email: string;
  password: string;
  expectedResult: 'success' | 'invalid_email' | 'invalid_password' | 'empty_fields';
  expectedMessage?: string;
}

// Use in tests
const loginScenarios: LoginTestData[] = [
  {
    email: 'valid@example.com',
    password: 'ValidPass123',
    expectedResult: 'success'
  },
  {
    email: 'invalid-email',
    password: 'pass123',
    expectedResult: 'invalid_email',
    expectedMessage: 'Please enter a valid email address'
  },
  {
    email: 'test@example.com',
    password: 'wrongpassword',
    expectedResult: 'invalid_password',
    expectedMessage: 'Incorrect password'
  }
];

// Run tests with data
loginScenarios.forEach(scenario => {
  test(`login with ${scenario.expectedResult}`, async ({ page }) => {
    await page.goto('/login');
    await page.getByLabel('Email').fill(scenario.email);
    await page.getByLabel('Password').fill(scenario.password);
    await page.getByRole('button', { name: 'Sign In' }).click();
    
    if (scenario.expectedResult === 'success') {
      await expect(page).toHaveURL('**/dashboard');
    } else {
      await expect(page.getByRole('alert')).toHaveText(scenario.expectedMessage!);
    }
  });
});

Async/Await & Promises

Async/await is essential for test automation - most operations (API calls, waiting for elements) are asynchronous.

Promises

// Promise represents a future value
const promise = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('Data loaded!');
    // or reject(new Error('Failed!'));
  }, 1000);
});

// Using promise
promise
  .then(data => console.log(data)) // "Data loaded!"
  .catch(error => console.error(error));

// Promise states
// - Pending: initial state
// - Fulfilled: operation completed successfully
// - Rejected: operation failed

Async/Await

// async function returns a Promise
async function fetchUser(id: number): Promise {
  const response = await fetch(`/api/users/${id}`);
  
  if (!response.ok) {
    throw new Error(`Failed to fetch user: ${response.status}`);
  }
  
  return response.json();
}

// Using async function
async function test() {
  try {
    const user = await fetchUser(123);
    console.log(user.name);
  } catch (error) {
    console.error('Failed to fetch user:', error);
  }
}

// Multiple async operations
async function loadData() {
  // Sequential (slower)
  const user = await fetchUser(1);
  const posts = await fetchPosts(user.id);
  
  // Parallel (faster) - Promise.all
  const [user, posts] = await Promise.all([
    fetchUser(1),
    fetchPosts(1)
  ]);
}

Common Patterns in Tests

// Wait for condition with timeout
async function waitForElement(locator: Locator, timeout = 5000): Promise {
  const start = Date.now();
  while (Date.now() - start < timeout) {
    if (await locator.isVisible()) return;
    await page.waitForTimeout(100);
  }
  throw new Error(`Element not visible after ${timeout}ms`);
}

// Retry pattern
async function retry(
  fn: () => Promise,
  maxAttempts = 3,
  delay = 1000
): Promise {
  let lastError: Error;
  
  for (let i = 0; i < maxAttempts; i++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error as Error;
      if (i < maxAttempts - 1) {
        await page.waitForTimeout(delay);
      }
    }
  }
  
  throw lastError!;
}

Type-Safe Page Objects

import { Page, Locator, expect } from '@playwright/test';

interface LoginPageOptions {
  baseURL: string;
  timeout: number;
}

export class LoginPage {
  readonly page: Page;
  readonly emailInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;
  readonly errorAlert: Locator;

  constructor(page: Page, options?: LoginPageOptions) {
    this.page = page;
    this.emailInput = page.getByLabel('Email');
    this.passwordInput = page.getByLabel('Password');
    this.submitButton = page.getByRole('button', { name: /sign in/i });
    this.errorAlert = page.locator('[role="alert"]');
  }

  async goto(baseURL: string = '/'): Promise {
    await this.page.goto(`${baseURL}/login`);
  }

  async login(credentials: {
    email: string;
    password: string;
  }): Promise {
    await this.emailInput.fill(credentials.email);
    await this.passwordInput.fill(credentials.password);
    await this.submitButton.click();
  }

  async expectErrorMessage(message: string): Promise {
    await expect(this.errorAlert).toHaveText(message);
  }
}
