Master the WebdriverIO automation framework for browser testing

What is WebdriverIO?

WebdriverIO is a popular test automation framework built on Node.js. It provides a powerful and flexible API for browser and mobile testing, with built-in support for Mocha, Jasmine, and Cucumber test runners.

Why WebdriverIO?

  • Simple Syntax: Easy to learn and write tests
  • Built-in Wait Logic: Automatic waiting for elements
  • Custom Commands: Extend functionality easily
  • Service Integrations: Appium, Sauce Labs, BrowserStack
  • Modern Architecture: Built for modern web apps
  • TypeScript Support: Full TypeScript support out of the box

Setup & Configuration

Quick Setup

# Create project directory
mkdir my-wdio-tests && cd my-wdio-tests

# Initialize npm project
npm init -y

# Install WebdriverIO CLI and test runner
npm install @wdio/cli

# Run setup wizard (creates wdio.conf.js)
npx wdio config

# Install browsers (if needed)
npx wdio install chromium

# Run tests
npx wdio run wdio.conf.js

Configuration File (wdio.conf.js)

export const config = {
    // Spec files pattern
    specs: ['./tests/**/*.spec.ts'],
    
    // Capabilities
    capabilities: [{
        maxInstances: 5,
        browserName: 'chrome',
        'goog:chromeOptions': {
            args: ['--headless', '--disable-gpu']
        }
    }],
    
    // Test runner
    runner: 'local',
    
    // Framework (mocha, jasmine, or cucumber)
    framework: 'mocha',
    
    // Reporter options
    reporters: ['spec'],
    
    // Mocha options
    mochaOpts: {
        ui: 'bdd',
        timeout: 60000
    },
    
    // Base URL for tests
    baseUrl: 'http://localhost:3000',
    
    // Auto-wait settings
    waitforTimeout: 5000,
    waitforInterval: 500,
    
    // Hooks
    before: () => { /* Before all tests */ },
    after: () => { /* After all tests */ },
    beforeTest: () => { /* Before each test */ },
    afterTest: () => { /* After each test */ }
};

Selectors

WebdriverIO provides multiple selector strategies to find elements on a page.

Selector Priority (Best to Worst)

// 1. Accessibility Selectors (Recommended)
$('~accessibility-id')           // iOS
$('android=UiSelector().xxx')    // Android

// 2. CSS Selectors (Preferred for web)
$('#submit-button')
$('.btn-primary')
$('input[type="email"]')
$('div.container > .card')

// 3. Link Text
$('=Sign In')                    // Partial: contains
$('*=ign')                       // Exact: equals

// 4. XPath
$('//button[contains(text(), "Submit")]')
$('//div[@id="form"]//input[@name="email"]')

// 5. Chain Selectors
const form = $('#login-form');
const emailInput = form.$('input[name="email"]');

Working with Elements

// Single element
const submitBtn = $('#submit-btn');
await submitBtn.click();

// Multiple elements (returns array)
const listItems = $$('.item');
for (const item of listItems) {
    console.log(await item.getText());
}

// Element state checks
await $('#btn').waitForDisplayed({ timeout: 5000 });
await $('#btn').waitForEnabled({ timeout: 5000 });
await $('#btn').waitForExist({ timeout: 5000 });

// Get element info
const text = await $('h1').getText();
const isVisible = await $('h1').isDisplayed();
const isEnabled = await $('input').isEnabled();
const tagName = await $('div').getTagName();
const attr = await $('a').getAttribute('href');

Waiting

// Built-in implicit wait (in config)
await browser.setTimeout({ 'implicit': 5000 });

// Explicit waits
await $('button').waitForDisplayed({ timeout: 5000, reverse: true });
await $('button').waitForEnabled({ timeout: 5000 });
await $('button').waitForExist({ timeout: 5000 });

// Custom wait
await browser.waitUntil(
    async () => (await browser.getTitle()) === 'Expected Title',
    { timeout: 5000, timeoutMsg: 'Title did not match' }
);

// Pause (avoid in real tests!)
await browser.pause(1000);

Assertions

Built-in Assertions (expect-webdriverio)

import expect from 'expect-webdriverio';

// Text assertions
await expect($('h1')).toHaveText('Welcome');
await expect($('h1')).not.toHaveText('Error');
await expect($('h1')).toContainText('Welcome');

// Value assertions
await expect($('input')).toHaveValue('default');
await expect($('input')).toHaveValueContaining('part');

// Visibility assertions
await expect($('button')).toBeDisplayed();
await expect($('button')).not.toBeDisplayed();
await expect($('button')).toBeVisible();
await expect($('button')).toBeHidden();

// Attribute assertions
await expect($('a')).toHaveAttr('href', 'https://example.com');
await expect($('input')).toHaveClass('active');
await expect($('input')).not.toHaveClass('error');

// State assertions
await expect($('input')).toBeEnabled();
await expect($('input')).toBeDisabled();
await expect($('input[type="checkbox"]')).toBeChecked();

Custom Assertions

// Add custom assertion
expect.extend({
    async toHaveCssProperty(received, property, value) {
        const actual = await received.getCSSProperty(property);
        const pass = actual.value === value;
        return {
            pass,
            message: () => pass
                ? `Expected element not to have CSS ${property}: ${value}`
                : `Expected element to have CSS ${property}: ${value}, but got: ${actual.value}`
        };
    }
});

// Use in test
await expect($('#box')).toHaveCssProperty('background-color', 'rgb(255, 0, 0)');

Page Objects

Login Page Object Example

// pageobjects/LoginPage.js
class LoginPage {
    // Selectors
    get emailInput() { return $('input[type="email"]'); }
    get passwordInput() { return $('input[type="password"]'); }
    get submitButton() { return $('button[type="submit"]'); }
    get errorMessage() { return $('.alert-danger'); }

    // Methods
    async open() {
        await browser.url('/login');
    }

    async login(email, password) {
        await this.emailInput.setValue(email);
        await this.passwordInput.setValue(password);
        await this.submitButton.click();
    }

    async getErrorMessage() {
        await this.errorMessage.waitForDisplayed();
        return this.errorMessage.getText();
    }
}

export default new LoginPage();

Using Page Objects in Tests

// specs/login.spec.js
import LoginPage from '../pageobjects/LoginPage';
import DashboardPage from '../pageobjects/DashboardPage';

describe('Login Flow', () => {
    it('should login with valid credentials', async () => {
        await LoginPage.open();
        await LoginPage.login('test@example.com', 'password123');
        
        // Assert redirect to dashboard
        await expect(browser).toHaveUrl('**/dashboard');
    });

    it('should show error with invalid credentials', async () => {
        await LoginPage.open();
        await LoginPage.login('invalid@example.com', 'wrongpass');
        
        // Assert error message
        const errorMsg = await LoginPage.getErrorMessage();
        await expect(errorMsg).toContain('Invalid credentials');
    });
});

Hooks & Lifecycle

Test Lifecycle Hooks

// wdio.conf.js
export const config = {
    // Runs before all tests start
    before: async () => {
        console.log('Starting test suite...');
    },

    // Runs after all tests finish
    after: async () => {
        console.log('Test suite completed.');
    },

    // Runs before each test file
    beforeTest: async (test, context) => {
        console.log(`Running: ${test.title}`);
    },

    // Runs after each test
    afterTest: async (test, context, { error, result, duration, passed, retries }) => {
        if (error) {
            console.log(`Test failed: ${test.title}`);
            // Take screenshot on failure
            await browser.saveScreenshot(`./error-${test.title}.png`);
        }
    },

    // Runs before each hook
    beforeHook: async () => {
        // Setup before each test/hook
    },

    // Runs after each hook
    afterHook: async () => {
        // Cleanup after each test/hook
    }
};

Custom Commands

// wdio.conf.js - Add to config
before: async () => {
    // Add custom command to browser object
    browser.addCommand('waitForElementVisible', async function(selector, timeout = 5000) {
        const element = await $(selector);
        await element.waitForDisplayed({ timeout });
        return element;
    });

    // Add command to element object
    addCommand('toBeClickable', async function(timeout = 5000) {
        await this.waitForDisplayed({ timeout });
        await this.waitForEnabled({ timeout });
        return this;
    });
};

// Use in test
await browser.waitForElementVisible('#submit-btn');
await $('#submit-btn').toBeClickable();
